1 Conversion Rate

Conversion Rate =
(Closed Won Deals / Total Closed Deals) × 100


SELECT
ROUND(
(
COUNT(*) FILTER (WHERE stage = 'CLOSED_WON') * 100.0 /
NULLIF(
COUNT(*) FILTER (
WHERE stage IN (
'CLOSED_WON',
'CLOSED_LOST',
'CLOSED_LOST_TO_COMPETITION'
)
),0
)
)::numeric,
2
) AS conversion_rate
FROM "Deal";


//////////////////////////////////

2 Revenue Won


Revenue Won = SUM(amount WHERE stage = CLOSED_WON)


SELECT
SUM(amount) AS revenue_won
FROM "Deal"
WHERE stage = 'CLOSED_WON';

/////////////////////////////////////

3 Win / Loss Ratio


Win / Loss Ratio =
Closed Won Deals : Closed Lost Deals


SELECT
COUNT(*) FILTER (WHERE stage = 'CLOSED_WON') AS won_deals,
COUNT(*) FILTER (
WHERE stage IN ('CLOSED_LOST','CLOSED_LOST_TO_COMPETITION')
) AS lost_deals
FROM "Deal";

//////////////////////////////////////////

4 Revenue Realization Rate

Revenue Realization Rate =
Revenue Won / (Revenue Won + Revenue Lost) × 100


SELECT
ROUND(
(
SUM(CASE WHEN stage = 'CLOSED_WON' THEN amount ELSE 0 END) * 100.0 /
NULLIF(
SUM(
CASE
WHEN stage IN (
'CLOSED_WON',
'CLOSED_LOST',
'CLOSED_LOST_TO_COMPETITION'
)
THEN amount ELSE 0
END
),0
)
)::numeric,
2
) AS revenue_realization_rate
FROM "Deal";

model EmailLog {
  id String @id @default(uuid())

  toEmail  String
  ccEmails String?
  bccEmails String?

  subject String
  body    String @db.Text

  status   EmailStatus   @default(SENT)
  provider EmailProvider?

  direction EmailDirection @default(OUTBOUND)

  messageId String?
  threadId  String?
  inReplyTo String?

  templateId String?
  template   EmailTemplate? @relation(fields: [templateId], references: [id])

  dealId String?
  deal   Deal? @relation(fields: [dealId], references: [id], onDelete: SetNull)

  contactId String?
  contact   Contact? @relation(fields: [contactId], references: [id], onDelete: SetNull)

  accountId String?
  account   Account? @relation(fields: [accountId], references: [id], onDelete: SetNull)

  sentById String
  sentBy   User @relation(fields: [sentById], references: [id])

  opened  Boolean @default(false)
  replied Boolean @default(false)
  clicked Boolean @default(false)

  sentAt DateTime?
  createdAt DateTime @default(now())

  @@index([dealId])
  @@index([contactId])
  @@index([accountId])
}


model EmailTemplate {
  id String @id @default(uuid())

  name     String
  subject  String
  body     String @db.Text

  category TemplateCategory?
  isActive Boolean @default(true)

  createdById String
  createdBy   User @relation(fields: [createdById], references: [id])

  emails EmailLog[]

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  @@index([name])
}
